/* styles.css */ .container { max-width: 600px; margin: 0 auto; padding: 2rem; font-family: Arial, sans-serif; } button { background-color: #28a745; color: white; border: none; padding: 10px 15px; cursor: pointer; font-size: 1.2em; } button:hover { background-color: #218838; } /* script.js */ const jokeButton = document.getElementById('jokeButton'); const jokeDisplay = document.getElementById('jokeDisplay'); jokeButton.addEventListener('click', () => { jokeDisplay.innerHTML = 'Loading...'; fetch('https://v2.jokeapi.dev/joke/Any') .then(response => { if (!response.ok) { throw new Error('Network response was not ok'); } return response.json(); }) .then(data => { if (data.type === 'single') { jokeDisplay.innerHTML = data.joke; } else { jokeDisplay.innerHTML = `${data.setup}
${data.delivery}`; } }) .catch(error => { jokeDisplay.innerHTML = 'Failed to fetch a joke. Please try again later.'; console.error('There was a problem with the fetch operation:', error); }); }); /* Add responsive design and accessibility features as needed */
